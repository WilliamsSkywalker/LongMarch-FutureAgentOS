编辑模式 vs. 创建模式

编辑模式不是 "重新创建"，而是 在现有代码上做修改。这要求：

  • 保留现有代码的结构和状态
  • 只修改与用户需求相关的部分
  • 失败时能恢复原始版本

源码结构

// EditMode.swift
// 职责：在现有应用上执行修改

struct EditMode {
    let strategy: ToolRepairStrategy
    let versionBackup: ToolVersionBackupClient
    let repairLoop: ContentViewBuildRepairLoop
    
    // 主入口：编辑现有应用
    func editTool(
        tool: Tool,
        modificationPrompt: String
    ) async -> EditResult {
        
        // 1. 读取当前代码
        let currentCode = readContentView(from: tool)
        
        // 2. 暂存当前版本（原子备份）
        versionBackup.stagePendingVersion(
            code: currentCode,
            for: tool
        )
        
        // 3. 根据策略选择编辑方式
        let editedCode: String
        
        switch strategy {
        case .diffRepair:
            // 模型支持 Diff：发送 bounded unified diff 请求
            editedCode = await editWithDiff(
                currentCode: currentCode,
                modification: modificationPrompt
            )
            
        case .fullRewrite:
            // 模型不支持 Diff：发送完整重写请求
            editedCode = await editWithFullRewrite(
                currentCode: currentCode,
                modification: modificationPrompt
            )
        }
        
        // 4. 编译验证（复用修复循环）
        let (finalCode, success) = await repairLoop.buildAndRepair(
            code: editedCode,
            packageName: tool.displayName
        )
        
        if success {
            // 5. 成功：提升暂存版本为 "previous"（备份旧版本）
            versionBackup.promotePendingToPrevious(for: tool)
            
            // 6. 写入最终代码
            writeContentView(finalCode, to: tool)
            
            return .success
        } else {
            // 7. 失败：从 pending 恢复原始代码
            let originalCode = versionBackup.restorePendingVersion(for: tool)
            writeContentView(originalCode, to: tool)
            
            versionBackup.discardPendingVersion(for: tool)
            
            return .failure(.editFailed)
        }
    }
    
    // MARK: - Diff 策略
    private func editWithDiff(
        currentCode: String,
        modification: String
    ) async -> String {
        
        let systemPrompt = """
        You are a precise code editor. Given a user's modification request
        and the current source code, generate a minimal unified diff
        that implements the change.
        """
        
        let userPrompt = """
        Current code (ContentView.swift):
        \(currentCode)
        
        User's modification request:
        \(modification)
        
        Generate a unified diff that implements this change.
        Rules:
        - Only modify ContentView.swift
        - Make minimal changes
        - Include 3 lines of context around each change
        - Do NOT change unrelated code
        """
        
        let diff = await languageModel.chat(
            system: systemPrompt,
            user: userPrompt
        )
        
        // 解析 diff 并应用到代码
        return applyDiff(diff: diff, to: currentCode)
    }
    
    // MARK: - 全量重写策略
    private func editWithFullRewrite(
        currentCode: String,
        modification: String
    ) async -> String {
        
        let systemPrompt = ToolGenerationPrompts.singleFileCodingInstructions
        
        let userPrompt = """
        You are editing an existing app. Here is the current code:
        
        \(currentCode)
        
        The user wants to modify it:
        \(modification)
        
        Return the COMPLETE updated ContentView.swift with the modification applied.
        Preserve all existing functionality that wasn't mentioned for modification.
        """
        
        return await languageModel.chat(
            system: systemPrompt,
            user: userPrompt
        )
    }
}

// 版本备份客户端
struct ToolVersionBackupClient {
    
    func stagePendingVersion(code: String, for tool: Tool) {
        let pendingPath = tool.versionsDirectory
            .appendingPathComponent("pending-ContentView.swift")
        writeFile(code, to: pendingPath)
    }
    
    func promotePendingToPrevious(for tool: Tool) {
        let pendingPath = tool.versionsDirectory
            .appendingPathComponent("pending-ContentView.swift")
        let previousPath = tool.versionsDirectory
            .appendingPathComponent("previous-ContentView.swift")
        
        // 如果已有 previous，删除旧的
        if fileExists(previousPath) {
            deleteFile(previousPath)
        }
        
        // 将 pending 提升为 previous
        moveFile(from: pendingPath, to: previousPath)
    }
    
    func restorePendingVersion(for tool: Tool) -> String {
        let pendingPath = tool.versionsDirectory
            .appendingPathComponent("pending-ContentView.swift")
        return readFile(pendingPath)
    }
    
    func discardPendingVersion(for tool: Tool) {
        let pendingPath = tool.versionsDirectory
            .appendingPathComponent("pending-ContentView.swift")
        deleteFile(pendingPath)
    }
    
    // 用户手动恢复上一版本
    func restorePreviousVersion(for tool: Tool) -> String {
        let previousPath = tool.versionsDirectory
            .appendingPathComponent("previous-ContentView.swift")
        let currentPath = tool.sourceFilePath
        
        let previousCode = readFile(previousPath)
        let currentCode = readFile(currentPath)
        
        // 交换：previous → current，current → previous
        writeFile(previousCode, to: currentPath)
        writeFile(currentCode, to: previousPath)
        
        return previousCode
    }
}

两种策略对比

维度 Diff 策略 全量重写策略
适用模型 支持长上下文 + 精确编辑（GPT-4o, Claude 3.5） 不支持 diff 或上下文较短的模型
Token 消耗 低（只返回修改部分） 高（返回完整文件）
上下文使用 高效（只传必要部分） 需要完整文件在上下文中
风险 Diff 解析可能失败 可能意外修改无关代码
验证 应用后编译，错误增加则回滚 直接编译，错误增加则回滚

策略选择：基于模型能力

这是 Ironsmith 的关键设计：不根据模型名称选择策略，而是根据模型能力。

// ToolRepairStrategy.swift
// 由 InferenceStore 从模型来源和能力推导

enum ToolRepairStrategy {
    case diffRepair       // 支持 unified diff
    case fullRewrite      // 只支持完整重写
    
    static func infer(from model: ModelConfig) -> ToolRepairStrategy {
        // 基于模型来源和能力判断
        switch model.providerIdentifier {
        case "openai", "anthropic":
            // GPT-4o, Claude 3.5 支持 diff
            return .diffRepair
            
        case "ollama", "mlx":
            // 本地模型通常不支持 diff
            return .fullRewrite
            
        case "apple.foundation":
            // Apple Foundation Model 能力最弱
            return .fullRewrite
            
        default:
            // 未知模型，保守选择
            return .fullRewrite
        }
    }
}

版本历史设计

Ironsmith 的版本历史不是 Git，而是 简单的前一版本备份：

// 版本文件结构
~/.ironsmith/tools/MyApp/.ironsmith/versions/
├── pending-ContentView.swift      // 编辑中的暂存版本
└── previous-ContentView.swift     // 上一次成功的版本（用于恢复）

// 编辑流程
[编辑前]  current → pending (暂存)
[编辑成功] pending → previous (旧版成为备份), current = 新版
[编辑失败] pending → current (恢复), discard pending

💡 设计哲学： 不要过度工程化。一个备份就够了。用户不需要完整的 Git 历史，只需要 "撤销最后一次编辑"。

对 LongMarch 的启示

LongMarch 的编辑模式应该：

  • 编辑前保存当前版本到数据库
  • 支持 "撤销编辑" 功能
  • 根据模型能力选择 diff 或全量重写
  • 不要追求完整的 Git 历史，一个备份就够了
