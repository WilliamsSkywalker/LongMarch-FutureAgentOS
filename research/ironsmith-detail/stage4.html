Harness（代码逻辑）· 第一道可靠性防线

职责

在 LLM 输出和编译之间，插入一个 100% 确定性的代码清洗层。这是 Ironsmith 可靠性的第一道防线。LLM 输出不可靠（即使系统提示禁止，LLM 仍有 5-10% 概率违反约束），但代码规则的 cleanup 是 100% 可靠的。

源码结构

// ContentViewSourceCleanup.swift
// 职责：清洗 LLM 输出，确保符合编译要求
// 关键：纯代码规则，不调用 LLM

struct ContentViewSourceCleanup {
    
    func clean(_ rawSource: String) -> String {
        var source = rawSource
        
        // 1. 剥离 Markdown 围栏和 think 块
        source = stripMarkdownFences(source)
        source = stripThinkBlocks(source)
        
        // 2. 移除 LLM 生成的 App/Preview 结构
        source = removeAppBlock(source)
        source = removePreviewBlock(source)
        
        // 3. 规范化 import 语句
        source = normalizeImports(source)
        
        // 4. 修复 macOS SwiftUI 常见陷阱
        source = fixCommonFootguns(source)
        
        // 5. 将松散顶层状态移入 ContentView
        source = moveLooseStateIntoContentView(source)
        
        // 6. 移除错位成员作用域视图块
        source = removeMisplacedViewBlocks(source)
        
        // 7. 尝试包裹松散的 SwiftUI 片段
        source = wrapLooseSwiftUIFragments(source)
        
        // 8. 运行 swift-format 统一格式
        source = formatWithSwiftFormat(source)
        
        return source
    }
    
    // MARK: - 具体规则实现
    
    private func stripMarkdownFences(_ source: String) -> String {
        // 移除 ```swift ... ``` 围栏
        var result = source
        let fences = ["```swift", "```", "```json"]
        for fence in fences {
            result = result.replacingOccurrences(of: fence, with: "")
        }
        return result.trimmingCharacters(in: .whitespacesAndNewlines)
    }
    
    private func stripThinkBlocks(_ source: String) -> String {
        // 移除  ...  或类似的 reasoning 块
        var result = source
        let patterns = [
            ".*?",           // 显式 think 标签
            "Thinking:.*?(?=\n\n|struct)",   // "Thinking:" 前缀
            "Let me think.*?(?=\nstruct)",  // 思考过程
        ]
        for pattern in patterns {
            result = result.replacingOccurrences(
                of: pattern,
                with: "",
                options: .regularExpression
            )
        }
        return result
    }

        // 移除 #Preview { ... } 或 struct Preview: PreviewProvider
        var result = source
        let previewPatterns = [
            "#Preview.*?\\{.*?\\}",
            "struct.*PreviewProvider.*\\{.*?\\}",
        ]
        for pattern in previewPatterns {
            result = result.replacingOccurrences(
                of: pattern,
                with: "",
                options: .regularExpression
            )
        }
        return result
    }
    
    private func normalizeImports(_ source: String) -> String {
        // 去重、排序 import 语句
        let importLines = source.components(separatedBy: .newlines)
            .filter { $0.hasPrefix("import ") }
        
        let uniqueImports = Array(Set(importLines)).sorted()
        
        // 移除原始 import，插入排序后的
        var result = source
        for line in importLines {
            result = result.replacingOccurrences(of: line + "\n", with: "")
        }
        
        // 在文件顶部插入排序后的 import
        if let firstLine = uniqueImports.first {
            result = uniqueImports.joined(separator: "\n") + "\n\n" + result
        }
        
        return result
    }
    
    private func fixCommonFootguns(_ source: String) -> String {
        var result = source
        
        // 修复 1：错误的 .frame 修饰符
        result = result.replacingOccurrences(
            of: ".frame(width: height:)",
            with: ".frame(width:height:)"
        )
        
        // 修复 2：错误的 List 初始化
        result = result.replacingOccurrences(
            of: "List(items) { item in",
            with: "List(items, id: \\.self) { item in"
        )
        
        // 修复 3：macOS 特有的 NavigationView vs NavigationStack
        result = result.replacingOccurrences(
            of: "NavigationStack {",
            with: "NavigationView {"
        )
        
        return result
    }
    
    private func moveLooseStateIntoContentView(_ source: String) -> String {
        // LLM 有时会把 @State 变量放在 struct 外面
        // 需要把它们移入 ContentView 内部
        var result = source
        
        // 找到 ContentView 的 body 之前的位置
        if let bodyRange = result.range(of: "var body: some View") {
            // 提取所有顶层 @State 声明
            let statePattern = "@State.*"
            let looseStates = result.matches(for: statePattern)
            
            // 移除顶层的，插入到 body 之前
            for state in looseStates {
                result = result.replacingOccurrences(of: state, with: "")
            }

        }
        
        return result
    }

清洗规则详解

规则 为什么需要 LLM 违反频率
剥离 Markdown 围栏 LLM 经常在代码外层加 ```swift ~30%
剥离 Think 块 reasoning 模型会输出思考过程 ~50%（reasoning 模型）
移除 Preview 块 LLM 训练数据中有大量 preview，习惯性添加 ~10%
移除 App 结构 LLM 想写 "完整应用"，会加 @main ~5%
修复修饰符陷阱 SwiftUI 修饰符语法容易出错 ~15%
移动松散状态 LLM 会把 @State 放在 struct 外面 ~8%

为什么 "纯代码规则" 比 "让 LLM 自己修正" 更可靠

有两个原因：

1. LLM 的 "修正" 可能引入新错误

如果让 LLM 自己修正错误（比如 "请移除 Preview 块"），LLM 可能：
- 移除了 Preview 块，但意外地删除了相邻的代码
- 添加了新的 import 来解决 "找不到 PreviewProvider" 的错误，但这个 import 本身有问题

2. 代码规则是 100% 确定性的

正则表达式和字符串替换不会 "理解" 代码，但它们也不会 "误解" 代码。如果规则是 "移除所有 ```swift 行"，它总是正确地移除这些行，不会误伤其他代码。

🔧 Harness 设计原则： 凡是能用代码规则解决的，绝不用 LLM。LLM 只在 "需要理解代码语义" 的场景使用（如修复编译错误）。
