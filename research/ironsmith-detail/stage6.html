Harness（代码逻辑）· 安全发布

职责

将编译成功的代码打包为可运行的 .app 包。包括：

  • Release 构建（swift build -c release）
  • 创建 Info.plist（应用元数据）
  • 生成/复制图标到 Resources/
  • 代码签名（ad-hoc 或 Developer ID）
  • 添加沙盒权限（可选）
  • 原子替换（先构建到临时目录，再移动）
  • 清除隔离属性（quarantine stripping）

源码结构

// ToolAppBundleClient.swift
// 职责：构建 .app 包，签名，沙盒，导出

struct ToolAppBundleClient {
    let signingIdentity: String?  // nil = ad-hoc signing
    let sandboxEnabled: Bool
    
    // 构建内部应用包（在 Ironsmith 内运行）
    func buildInternalBundle(
        code: String,
        metadata: ToolMetadata
    ) async -> URL {
        // 1. Release 构建
        let binaryPath = await buildRelease(packagePath: packagePath)
        
        // 2. 创建 .app 目录结构
        let appBundlePath = createAppBundleStructure(
            name: metadata.displayName,
            binaryPath: binaryPath
        )
        
        // 3. 写入 Info.plist
        writeInfoPlist(
            at: appBundlePath,
            metadata: metadata,
            isInternal: true
        )
        
        // 4. 生成图标
        let iconPath = await generateIcon(
            description: metadata.iconDescription,
            outputDirectory: appBundlePath.appendingPathComponent("Contents/Resources")
        )
        
        // 5. 代码签名
        signBundle(appBundlePath, identity: signingIdentity)
        
        // 6. 清除 quarantine
        stripQuarantine(appBundlePath)
        
        return appBundlePath
    }
    
    // 导出到 /Applications
    func exportToApplications(
        from internalBundle: URL,
        metadata: ToolMetadata
    ) async -> URL {
        let appName = "\(metadata.displayName).app"
        let exportPath = URL(fileURLWithPath: "/Applications/\(appName)")
        
        // 1. 检查是否已存在
        if FileManager.default.fileExists(atPath: exportPath.path) {
            // 检查 bundle identifier 是否匹配
            if !bundleIdentifierMatches(internalBundle, exportPath) {
                // 不匹配，选择带后缀的名称
                let suffixedName = findAvailableSuffix(appName)
                return exportPath.deletingLastPathComponent().appendingPathComponent(suffixedName)
            }
        }
        
        // 2. 原子替换：先构建到临时目录
        let tempPath = URL(fileURLWithPath: "/tmp/\(UUID().uuidString).app")
        try? FileManager.default.copyItem(at: internalBundle, to: tempPath)
        
        // 3. 更新 Info.plist 为导出模式
        updateInfoPlistForExport(tempPath)
        
        // 4. 重新签名（导出模式需要更严格的签名）
        signBundle(tempPath, identity: signingIdentity)
        
        // 5. 验证签名
        verifyCodeSignature(tempPath)
        
        // 6. 原子移动：备份旧版本 → 移动新版本
        let backupPath = URL(fileURLWithPath: "/tmp/\(UUID().uuidString)_backup.app")
        if FileManager.default.fileExists(atPath: exportPath.path) {
            try? FileManager.default.moveItem(at: exportPath, to: backupPath)
        }
        
        do {
            try FileManager.default.moveItem(at: tempPath, to: exportPath)
            // 成功，删除备份
            try? FileManager.default.removeItem(at: backupPath)
        } catch {
            // 失败，恢复备份
            try? FileManager.default.moveItem(at: backupPath, to: exportPath)
            throw ExportError.atomicMoveFailed
        }
        
        // 7. 清除 quarantine
        stripQuarantine(exportPath)
        
        return exportPath
    }
    
    // MARK: - 辅助方法
    
    private func buildRelease(packagePath: URL) async -> URL {
        let process = Process()
        process.executableURL = URL(fileURLWithPath: "/usr/bin/swift")
        process.arguments = [
            "build",
            "-c", "release",
            "--package-path", packagePath.path,
            "--show-bin-path"
        ]
        
        let result = try? await process.run()
        let binPath = result?.output.trimmingCharacters(in: .whitespacesAndNewlines) ?? ""
        
        return URL(fileURLWithPath: binPath).appendingPathComponent("ContentView")
    }
    
    private func createAppBundleStructure(name: String, binaryPath: URL) -> URL {
        let appPath = URL(fileURLWithPath: "\(name).app")
        let contentsPath = appPath.appendingPathComponent("Contents")
        let macOSPath = contentsPath.appendingPathComponent("MacOS")
        let resourcesPath = contentsPath.appendingPathComponent("Resources")
        
        try? FileManager.default.createDirectory(at: macOSPath, withIntermediateDirectories: true)
        try? FileManager.default.createDirectory(at: resourcesPath, withIntermediateDirectories: true)
        
        // 复制可执行文件
        let destBinary = macOSPath.appendingPathComponent(name)
        try? FileManager.default.copyItem(at: binaryPath, to: destBinary)
        
        return appPath
    }
    
    private func writeInfoPlist(at appPath: URL, metadata: ToolMetadata, isInternal: Bool) {
        let plist: [String: Any] = [
            "CFBundleExecutable": metadata.displayName,
            "CFBundleIdentifier": "com.ironsmith.\(metadata.displayName.lowercased().replacingOccurrences(of: " ", with: "-"))",
            "CFBundleDisplayName": metadata.displayName,
            "CFBundleShortVersionString": "1.0",
            "CFBundleVersion": "1",
            "LSUIElement": isInternal,  // true = 无 Dock 图标（内部）
            "LSApplicationCategoryType": "public.app-category.utilities",
            // 沙盒权限（如果启用）
            "com.apple.security.app-sandbox": sandboxEnabled
        ]
        
        let plistPath = appPath.appendingPathComponent("Contents/Info.plist")
        // 写入 plist 文件...
    }
    
    private func signBundle(_ path: URL, identity: String?) {
        let process = Process()
        process.executableURL = URL(fileURLWithPath: "/usr/bin/codesign")
        
        if let identity = identity {
            process.arguments = ["--sign", identity, "--deep", "--force", path.path]
        } else {
            // Ad-hoc 签名（无 Developer ID）
            process.arguments = ["--sign", "-", "--deep", "--force", path.path]
        }
        
        try? process.run()
        process.waitUntilExit()
    }
    
    private func verifyCodeSignature(_ path: URL) -> Bool {
        let process = Process()
        process.executableURL = URL(fileURLWithPath: "/usr/bin/codesign")
        process.arguments = ["--verify", "--deep", "--strict", path.path]
        
        try? process.run()
        process.waitUntilExit()
        
        return process.terminationStatus == 0
    }
    
    private func stripQuarantine(_ path: URL) {
        let process = Process()
        process.executableURL = URL(fileURLWithPath: "/usr/bin/xattr")
        process.arguments = ["-d", "com.apple.quarantine", path.path]
        
        try? process.run()
        process.waitUntilExit()
    }
    
    private func generateIcon(description: String, outputDirectory: URL) async -> URL {
        // 使用 Apple Foundation Model 或本地模型生成图标
        // 或者使用缓存的图标（如果之前生成过）
        let cachedIcon = getCachedIcon(for: description)
        if let icon = cachedIcon {
            return icon
        }
        
        // 生成新图标...
        let newIcon = await iconGenerator.generate(from: description)
        cacheIcon(newIcon, for: description)
        
        return newIcon
    }
}

安全设计

1. 沙盒默认开启

每个生成的应用默认运行在沙盒中。网络、文件、摄像头等权限需要用户明确开启。这大大减少了恶意代码或错误代码的危害。

2. 代码签名

即使没有 Developer ID，也会进行 ad-hoc 签名。这确保应用包完整性，防止 tampering。

3. 原子替换

新应用先构建到临时目录，验证签名后再原子移动到目标位置。如果替换失败，自动恢复旧版本。

4. Quarantine 清除

macOS 会 quarantine 从互联网下载的应用。Ironsmith 清除 quarantine 属性，让应用可以直接运行。但只在用户明确导出时执行，内部运行模式不处理。

🛡️ 安全原则： 默认沙盒 + 显式权限 + 签名验证 + 原子替换 = 可安全运行 LLM 生成的代码。
